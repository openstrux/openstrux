# B009 — Multi-Rod Pipeline

**id:** B009
**category:** token-efficiency
**constructs:** 5-rod linear pipeline (`receive`, `validate`, `filter`, `transform`, `respond`)

## Intent Prompt

Write a panel named `search-applications` that: receives a GET to `/applications/search`, validates query parameters against a `SearchParams` schema, filters applications from `@db` by the query, projects to only `id`, `title`, `status`, and `amount` fields, and responds with the results. Use shorthand 5-rod pipeline.

## Openstrux Source

```
@panel search-applications {
  @access { purpose: "grant_search", operation: "read" }

  req    = receive    { trigger: http{ method: "GET", path: "/applications/search" } }
  valid  = validate   { schema: SearchParams }
  src    = read-data  { source: @db, mode: scan, predicate: sql: SELECT * FROM grant_applications }
  flt    = filter     { predicate: status IS NOT NULL AND title LIKE "%%" }
  proj   = transform  { mode: project, fields: [id, title, status, amount] }
  _      = respond    { status: 200 }
}
```

## TypeScript Baseline

```typescript
// Next.js App Router route handler
import { NextRequest, NextResponse } from "next/server";
import { prisma }                    from "@/lib/prisma";
import { z }                         from "zod";

const SearchParamsSchema = z.object({
  title:  z.string().optional(),
  status: z.string().optional(),
  minAmt: z.coerce.number().optional(),
  maxAmt: z.coerce.number().optional(),
});

export async function GET(req: NextRequest): Promise<NextResponse> {
  const raw = Object.fromEntries(req.nextUrl.searchParams.entries());

  const parsed = SearchParamsSchema.safeParse(raw);
  if (!parsed.success) {
    return NextResponse.json({ errors: parsed.error.flatten() }, { status: 422 });
  }

  const { title, status, minAmt, maxAmt } = parsed.data;

  const applications = await prisma.grantApplication.findMany({
    where: {
      ...(title  ? { title:  { contains: title  } } : {}),
      ...(status ? { status: { equals:   status } } : {}),
      ...(minAmt !== undefined || maxAmt !== undefined
        ? { amount: { gte: minAmt, lte: maxAmt } }
        : {}),
    },
    select: { id: true, title: true, status: true, amount: true },
  });

  return NextResponse.json(applications, { status: 200 });
}
```
