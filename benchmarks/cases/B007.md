# B007 — Context Inheritance

**id:** B007
**category:** token-efficiency
**constructs:** `strux.context`, `@context` inheritance

## Intent Prompt

Define a `strux.context` file that sets project-wide data protection defaults (controller "NLnet Foundation", controller_id "NL-2024", dpo "dpo@nlnet.nl"), a named Postgres source `@db`, and `@ops` (retry 3, timeout "30s"). Then write a panel that inherits all of it without repeating config.

## Openstrux Source

```
// strux.context
@context {
  @dp {
    controller:    "NLnet Foundation",
    controller_id: "NL-2024",
    dpo:           "dpo@nlnet.nl",
    record:        true,
    basis:         "legitimate_interest"
  }
  @source db = db.sql.postgres {
    host:        env("DB_HOST"),
    port:        5432,
    db_name:     env("DB_NAME"),
    credentials: secret_ref { provider: vault, path: "db/prod" }
  }
  @ops { retry: 3, timeout: "30s" }
}
```

```
// pipelines/applications/list.strux
@panel list-applications {
  @access { purpose: "grant_admin", operation: "read" }

  src = read-data { source: @db, mode: scan }
  _   = respond   { status: 200 }
}
```

```typescript
// lib/retry.ts — retry wrapper, referenced in each handler
export async function withRetry<T>(
  fn: () => Promise<T>,
  retries = 3,
  timeoutMs = 30_000
): Promise<T> {
  for (let attempt = 0; attempt < retries; attempt++) {
    const result = await Promise.race([
      fn(),
      new Promise<never>((_, reject) =>
        setTimeout(() => reject(new Error("timeout")), timeoutMs)
      ),
    ]);
    return result as T;
  }
  throw new Error("Max retries exceeded");
}
```

```typescript

import { NextResponse }   from "next/server";
import { prisma }         from "@/lib/db";
import { withRetry }      from "@/lib/retry";
import { DP_CONFIG }      from "@/lib/dp-config";

export async function GET(): Promise<NextResponse> {
  const applications = await withRetry(() =>
    prisma.grantApplication.findMany()
  );
  return NextResponse.json(applications, { status: 200 });
}
```
