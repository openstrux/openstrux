# B008 — Named Source Reference

**id:** B008
**category:** token-efficiency
**constructs:** named `@source` reference vs inline connection config

## Intent Prompt

Write a panel named `sync-to-analytics` that reads grant applications from a production Postgres database named `@production` and writes them to a BigQuery analytics target named `@analytics`, both defined in `strux.context`. Compare to writing the connection config inline in each panel.

## Openstrux Source

```
// strux.context (shared)
@context {
  @source production = db.sql.postgres {
    host:        env("DB_HOST"),
    port:        5432,
    db_name:     env("DB_NAME"),
    credentials: secret_ref { provider: vault, path: "db/prod" }
  }
  @target analytics = db.sql.bigquery {
    project:     env("BQ_PROJECT"),
    dataset:     "grants",
    location:    "EU",
    credentials: adc {}
  }
}
```

```
// pipelines/sync.strux
@panel sync-to-analytics {
  @access { purpose: "analytics_export", operation: "read" }

  src  = read-data  { source: @production, mode: scan }
  _    = write-data { target: @analytics }
}
```

## TypeScript Baseline

```typescript
// lib/connections.ts — inline connection config duplicated per pipeline
import { prisma } from "@/lib/prisma";
import { BigQuery } from "@google-cloud/bigquery";

export function getProductionDb() {
  return prisma; // wraps: host=process.env.DB_HOST, port=5432, db=process.env.DB_NAME
}

export function getAnalyticsBigQuery() {
  return new BigQuery({
    projectId: process.env.BQ_PROJECT,
    location: "EU",
    // credentials via ADC
  });
}
```

```typescript
// scripts/sync-to-analytics.ts
import { getProductionDb, getAnalyticsBigQuery } from "@/lib/connections";

export async function syncToAnalytics(): Promise<void> {
  const db  = getProductionDb();
  const bq  = getAnalyticsBigQuery();
  const dataset = bq.dataset("grants");

  const applications = await db.grantApplication.findMany();

  const table = dataset.table("grant_applications");
  await table.insert(
    applications.map((app) => ({
      id:          app.id,
      title:       app.title,
      amount:      app.amount,
      status:      app.status,
      submittedAt: app.submittedAt.toISOString(),
      applicantId: app.applicantId,
    }))
  );
}
```
